import { NextResponse } from 'next/server';
import nodemailer from 'nodemailer';

export async function POST(request: Request) {
  const { name, email, company, items, message } = await request.json();

  // Configure your email transporter
  const transporter = nodemailer.createTransport({
    service: 'gmail', // or your SMTP provider
    auth: {
      user: process.env.EMAIL_USER,
      pass: process.env.EMAIL_PASS,
    },
  });

  // Format the quote email
  const emailContent = `
    New Quote Request from ${name} (${company})
    Email: ${email}
    
    Items Requested:
    ${items.map((item: any) => `- ${item.name} x ${item.quantity}`).join('\n')}
    
    Message: ${message}
  `;

  await transporter.sendMail({
    from: process.env.EMAIL_USER,
    to: 'bongani@educore.co.za', // Your email
    subject: `New Quote Request from ${company}`,
    text: emailContent,
  });

  return NextResponse.json({ success: true });
}

// store/quoteStore.ts
import { create } from 'zustand';

interface QuoteItem {
  id: number;
  name: string;
  slug: string;
  sku: string;
  quantity: number;
  unit: string;
}

interface QuoteStore {
  items: QuoteItem[];
  addItem: (product: any, quantity: number) => void;
  removeItem: (id: number) => void;
  updateQuantity: (id: number, quantity: number) => void;
  clearQuote: () => void;
  getTotalItems: () => number;
}

export const useQuoteStore = create<QuoteStore>((set, get) => ({
  items: [],
  
  addItem: (product, quantity) => {
    set((state) => {
      const existingItem = state.items.find(item => item.id === product.id);
      if (existingItem) {
        return {
          items: state.items.map(item =>
            item.id === product.id
              ? { ...item, quantity: item.quantity + quantity }
              : item
          ),
        };
      }
      return {
        items: [...state.items, {
          id: product.id,
          name: product.name,
          slug: product.slug,
          sku: product.sku,
          quantity: quantity,
          unit: product.unit,
        }],
      };
    });
  },

  removeItem: (id) => {
    set((state) => ({
      items: state.items.filter(item => item.id !== id),
    }));
  },

  updateQuantity: (id, quantity) => {
    set((state) => ({
      items: state.items.map(item =>
        item.id === id ? { ...item, quantity } : item
      ),
    }));
  },
